289               }

291               else
292               {
293                  latitude=latitude + " N" ;
294               }
295               longitude = myXMLReader.getField("longitude");
296            //if longitude is negative, it is in the West,
297               if(longitude.substring(0,1).compareTo("-")==0)

299                  longitude=longitude.substring(1)+ " W" ;
300               }

303               {
304                  longitude=longitude + " E" ;
305               }

541        This function is used by the constructor to create and position all the various small

544        @param fmLeft an integer representing the number of pixels from the left of the frame
545        @param fmTop an integer representing the number of pixels from the top of the frame

551      JLabel label = new JLabel(strVar) ;                   //create the label
552      label.setFont(new Font("Helvetica", Font.BOLD,12)) ;  //set the font

554      label.setLocation(fmLeft,fmTop) ;                     //from left, from top

557      {
558         //BorderFactory pg 817
559         label.setBorder(BorderFactory.createLineBorder(Color.black,1)) ;
560      }
561      panel.add(label) ;                                    //add the label to the panel

563   }
