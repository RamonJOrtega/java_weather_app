011public class NWSFB               //name of the class
012{
013        private String strWeather ;   //the private class level variable only NWSFB can see
014        
015   /**
016   This construtor function sets a line of airport weather to the private class
017   variable, strWeather, for use in this class.
018   
019   @param strVar a line of weather for an airport
020   @return nothing   */
021        public NWSFB(String strVar)
022        {
023                strWeather = strVar ;
024        }
025        
026        public String getStationID(String strAlt)
027        {
028                return strAlt.substring(0,3) ;
029        }
030        /**
031   This private function accepts a string called strAlt representing an 
032   altitude and returns an integer representing the starting
033   character position of the altitude weather in the station weather string
034   by use of a switch statement for all of the 9 altitudes
035   
036   @param strAlt Altitude in thousands as a string
037   @return an integer representing the character postion with the strWeather string
038   */
039        private int getPos(String strAlt)
040        {
041                int intRet = 0 ;
042
043      switch (strAlt)
044      {
045        case "03" :
046            intRet = 4;
047            break;
048         case "06" :
049            intRet = 9;
050            break;
051         case "09" :
052            intRet = 17;
053            break;
054         case "12" :
055            intRet = 25;
056            break;
057         case "18" :
058            intRet = 33;         
059            break;
060         case "24" :
061            intRet = 41;
062            break;
063         case "30" :
064            intRet = 49;
065            break;
066         case "34" :
067            intRet = 56;
068            break;   
069         case "39" :
070            intRet = 63;
071            break;
072         default :
073            intRet = 4;
074            break;
075      }
076      return intRet;
077        }
078        /**
079   This public function will call getpos to determine where to 
080   start extracting seven characters from the class level variable
081        containing the station weather.
082   
083   @param strAlt - string representing altitude in thousands of feet.
084   @return seven character string representing the altitude weather.
085   */
086        public String getAltitudeWeather(String strAlt)
087        {
088                String altitudeWeather = "XX";
089                int intPos = getPos(strAlt);
090                
091                if (strAlt.compareTo("03") == 0 )
092                {
093        altitudeWeather = strWeather.substring(intPos, intPos + 5) ;
094                }
095                else
096                {
097        altitudeWeather = strWeather.substring(intPos, intPos + 7) ;
098                }
099                
100                return altitudeWeather ;
101        }
102        /**
103   This function will call getAltitudeWeather and extract the
104   first two characters and concatenate a zero to the
105   return value to show the user the direction in degrees that the
106   wind is coming from
107   
108   @param strAlt String representing altitude in thousands of feet.
109   @return windDir String representing wind direction at the specified altitude in degrees.
110   */
111        public String getWindDirection(String strAlt)
112        {
113                String windDir = "XX" ; //declare and initialize a wind direction variable string
114      if(getAltitudeWeather(strAlt).substring(0,2).compareTo("  ")== 0)
115      {//if the first two spaces of the altitude weather are blank, there is no wind direction
116         windDir = "N/A" ;
117      }
118      else if (getAltitudeWeather(strAlt).substring(0,4).compareTo("9900")== 0)
119      {//if the first for spaces of the altitude weather are 9900, the wind is calm
120        windDir = "Calm" ;
121      }
122      else if(Integer.parseInt(getAltitudeWeather(strAlt).substring(0,2)) > 36)
123      {//if the first two spaces of the altitude weather are greater than 36, subtract 50 from the direction
124         windDir = Integer.parseInt(getAltitudeWeather(strAlt).substring(0,2)) - 50 + "0 degrees";
125      }
126      else 
127      {//otherwise the wind direction is the first two characters of the altide weather times 10
128         windDir = getAltitudeWeather(strAlt).substring(0,2)+ "0 degrees" ;
129      }   
130      return windDir;
131        }
132         /**
133   This calls the method, getAltitudeWeather extract the 3nd and 4th
134   characters of the strWeather string. The method returns them as
135   the wind speed at the specified altitude.
136   
137   @param strAlt Altitude in thousands as a string
138   @return windSpeed The wind speed at the specified altitude in knots
139   */
140        public String getWindSpeed(String strAlt)
141        {
142                String windSpeed = "XX" ; //declare/initialize a windSpeed string variable
143      if(getAltitudeWeather(strAlt).substring(2,4).compareTo("  ") == 0)
144      {//if the 3rd and 4th character of strWeather are blank, there is no wind speed
145         windSpeed = "N/A" ;
146      }
147      else if (getAltitudeWeather(strAlt).substring(0,4).compareTo("9900") == 0)
148      {//if the first four characters of strWeather are 9900, the wind speed is calm or 0 knots
149         windSpeed = "Calm" ;      
150      }
151      else if (Integer.parseInt(getAltitudeWeather(strAlt).substring(0,2)) > 36)
152      {//if the 3rd and 4th character of strWeather are a number greater than 36, the wind speed is that number plus 100
153         windSpeed = Integer.parseInt(getAltitudeWeather(strAlt).substring(2,4)) + 100 + " knots" ;
154     }
155      else 
156      {// otherwise the windspeed is always the 3rd and 4th characters of the strWeather string
157         windSpeed = getAltitudeWeather(strAlt).substring(2,4)+ " knots" ;
158      }
159      return windSpeed;
160        }
161         /**
162   This public method calls the method, getAltitudeWeather to extract final characters
163   of the stationWeather string. The method returns them as
164   the temperature at the specified altitude.
165   
166   @param strAlt Altitude in thousands as a string
167   @return windTemp The temperature of the air at the specified altitude
168   */
169        public String getWindTemp(String strAlt)
170        {
171                String windTemp = "XX" ; //declare/initialize a windTemp varable string
172      if (getAltitudeWeather(strAlt).substring(4,5).compareTo(" ") == 0 )
173      {//if the character at position 4 is blank in the strWeather string, there is no wind temperature
174         windTemp = "N/A" ;
175      }//if the altitude we want temperature for is greater than 24000 feet, the temperature is always negative
176      else if (Integer.parseInt(strAlt) > 24 )
177      {
178         windTemp = "-" + getAltitudeWeather(strAlt).substring(4,6) + " degrees Celsius" ;
179      }
180      else
181      {//otherwise the wind temperature is a alwayst the last 3 characters of strWeather in degrees C
182         windTemp = getAltitudeWeather(strAlt).substring(4,7) + " C" ;
183      }// return the wind temperature string
184      return windTemp ;
185        }
186} // end of class
