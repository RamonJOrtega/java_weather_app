001import java.util.Scanner ; // Needed for Scanner class
002import java.io.* ;         // Needed for File and IOException

004/**
005   This program reads an input file called Ch01-text.txt, formats the file with tabs, and writes formated lines 
006   to an output file called Ch01-tab.txt
007
008    @author Ramon J. Ortega (Email: <a href="ramon.justis.ortega@gmail.com"> ramon.justis.ortega@gmail.com</a>)
009    @version 1.01 09/13/2013
010    @assignment.03 Quiz Reader App - A190-03
011    @prgm.usage Called from the operating system
012    @see "Gaddis, 2013, Starting Out with Java, From Control Structures through Objects, 5th Ed."
013    @see <a href="http://java.sun.com/javase/6/docs/technotes/guides/javadoc/index.html">JavaDoc Documentation</a>  
014*/
015
016public class A19003A
017{
018   public static void main(String[] args) throws IOException
019   {
020      File myFile = new File("Ch01-text.txt") ;       //Open inpute file
021       
022      if (myFile.exists() == true)                    //Does file even exist? If it does then we execute main program
023
024      {
025         //Declare variables and objects
026         int count = 1 ;                              //Make a counter to keep track of which question we are on
027         System.out.println("The file exists!") ;     //Tell the user if the input file exists
028         Scanner inputFile = new Scanner(myFile) ;    //Create a scanner object that will contain the input file 
029         String line = inputFile.nextLine();          //Create a string to contain 1 line of the inpute file
030                                                     
031         PrintWriter outputFile = new PrintWriter("Ch01-tab.txt");   //create the output file to write to
032        
033         for (int i=1 ; i<5 ; i++)                          //ignore/use up the first 5 lines of the text file
034         {
035            line = inputFile.nextLine() ;
036         }
037         while (inputFile.hasNext())                        //This loop iterates until the end of the file
038         {
039            line = inputFile.nextLine() ;                   //Read the file
040            line = line.trim() ;                            //Trim the spaces off of the front of every line
041            if (line.length() > 1)                          //If the line isn't all whitespace, begin comparing
042            {
043               if (line.substring(0,1).compareTo("_") == 0) //If the line is a question, begin writing to file
044               {
045                  System.out.println("question " + line) ;  //Display to the user that the line is a question
046                  if (count != 1)
047                  { 
048                     outputFile.println("") ;                //Print blank to the new file so seperate questions appear on different lines, except for the first time
049                  }
050                  if (count < 10)                           //All questions before number 10 must be printed starting at index 7
051                  {
052                  outputFile.print( "MC\t" + line.substring(7));  //Write the multiple choice question to the file line
053                  }
054                  else                                      //All questions after number 10 must be printed starting at index 8
055                  {                                                               
056                  outputFile.print( "MC\t" + line.substring(8));  //Write the multiple choice question to the file line
057                  }
058                  count = count + 1 ;                       //Keep track of which question we are on so we know where to start substrings
059               }
060               if (line.substring(1,2).compareTo(".") == 0) //If the line is an answer, begin writing to file
061               {
062                  System.out.println("answer " + line) ;    //Dispay  to the user that the line is an answer     
063                  outputFile.print( "\t" + line.substring(3)); //Write the answer to the file line
064                  outputFile.print( "\tIncorrect");            //Write that the answer is incorrect after the answer
065               }
066               if (line.equals("Answer Page Reference"))    //At this line of the input file,we know we found the answer key
067               {   
068                  while (!(line.equals("Starting Out with Visual Basic 2012, 6thEd by Tony Gaddis/Kip Irvine")))
069                  {                                         //Until we read the book title line, we are in the answer key
070                     line = inputFile.nextLine();           //Get the next line of the key   
071                     line = line.trim() ;                   //Get rid of the whitespace in the key
072                     System.out.println ("key" + line) ;    //Display to the user the lines that are part of the answer key
073                  }
074               }                
075            }
076            else 
077            {
078               System.out.println("not important" + line) ;    //Display what lines aren't going to be written to file
079            }
080
081         }  
082         inputFile.close();   //Always close the input file
083         outputFile.close();  //Always close the outpuf file
084      }
085      else
086      {
087         System.out.println("The file doesn't even exist!") ;  //Tell the user if the input file doesn't exist
088      }    
089   }
090}
