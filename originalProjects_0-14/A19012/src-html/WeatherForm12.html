202   This function is loads the cboApt ComboBox with the airport names from the stations table in the

210   public void btnLoadStations()

212      //create a string to hold the URL of the airport file, stations.txt
213      String URL = "http://faculty.sdmiramar.edu/jcouture/2013fl/cisc190/webct/manual/stations.txt";
214      //create a string to hold the name of the file
215      String myFile = "stations.txt" ;
216      //create a temporary string to hold the contents of the page on the URL
217      String strVar ;
218      //create an Inet object to put the stations.txt file contents into a string
219      Inet loadNet = new Inet() ;
220      strVar = loadNet.getFromFile(myFile) ;
221      //if the file is empty, go online and get page to create the file
222      if (strVar.compareTo("")==0 )
223      {
224         //get the weather page
225         strVar = loadNet.getURLRaw(URL) ;
226         //create the file
227         loadNet.saveToFile(myFile, strVar) ;
228      }

230      {
231         System.out.println("File Already On Disk") ;
232      }

234      File readFile = new File(myFile) ;
235        
236      DBUpdt db = new DBUpdt() ;          //create a DBUpdt object called db
237      db.openConnection("Weather") ;      // name of connection, not of file
238      db.deleteAll("stations") ;          //wipe the rows of the stations table clean everytime
239         
240   //Check to see if the stations.txt file already exists, if it does continue
241      if (readFile.exists())
242      {
243      //if the file exists do, the following reading of the txt file
244         try
245         {
246            Scanner inputFile = new Scanner(readFile) ;     //create a scanner object to read the file
247            StringBuilder stations = new StringBuilder(""); //create a stringBuilder object to store tokens in

249            while (inputFile.hasNext())                     //read every line of the World.txt file

251               String line = inputFile.nextLine();    
252            //Tokenize the line from the stations.txt file      
253               String[] tokens = line.split(",",3); // the tokens are separated by comas

257               db.setField("stations", "stationID" , tokens[0], "stationName", loadNet.properCase(tokens[1])) ;

270   /**
271   This function is loads the winds aloft weather from the NWS website into a text tile using the Inet class.
272   The text file then is read using the Inet class, and the station weather for each station is read from the file
273   and put into the Weather.mdb database using the DBUpdt class.
274   
275   @param none
276   */   
277   public void btnLoadWeather()
278   {
279      //Create a icaocode String to be used as a default if no other icaoCodes are found
280      String icaoCode = "KSAN" ; //default, but will be changed
281      //Create a string to hold the name of the winds aloft data file
282      String inputFile = "weather.txt" ;
283      //Create a string to hold the winds aloft URL
284      String windWeatherURL = "http://www.aviationweather.gov/products/nws/all" ;
285      try
286      {  //create a file to be read
287         File myFile = new File(inputFile) ; //pg 246
288         Scanner inFile = new Scanner(myFile);
289         //create a DBUpdt object to connect to the database
290         DBUpdt weatherLoad = new DBUpdt() ;          //create a DBUpdt object called weatherLoad
291         weatherLoad.openConnection("Weather") ;      // name of connection, not of file  
292         
293         while( inFile.hasNext() )                              //read the file line by line until the end
294         {
295            String line = inFile.nextLine() ;//store the next line of the file in a string variable
296                //only lines of greater than 65 characters are airport weather lines
297            if ( (line.length() > 65) && (line.substring(2,3).compareTo(" ") != 0) ) //if the next line of the file is an airport weather line
298            {
299               //use the DBUpdt object to put each line of weather from the weather.txt file into the database for each airport
300               weatherLoad.setField("stations", "stationID" , "K"+line.substring(0,3), "windsaloft", line) ;
301            }                                                                                   
302         }
303         weatherLoad.close() ;// close the database so others can use it
304         System.out.println("uploaded weathers") ;      //tell the user that the weather was updated
305      }
306      catch (FileNotFoundException e)
307      {
308         //if the weather.txt file does not yet exist, go online and download the data to create the file
309         //using the Inet class method: getPreData
310         //Display the exception to the user
311         System.out.println(e.getMessage()) ;
312         //Tell the user that the weather.txt file doesn't yet exist
313         System.out.println("The file " + inputFile + " does not yet exist" ) ;
314         //Tell the user that the winds aloft data will be downloaded
315         System.out.println("Downloading NCEP Winds Aloft Forecasts from the internet" );
316         //Create an Inet object to store webpage data in a string
317         Inet net = new Inet() ;
318         String rawWebPage ;
319         rawWebPage = net.getURLRaw(windWeatherURL) ;
320         //create a string to put the preData in 
321         String preData ;
322         preData = net.getPREData(rawWebPage);
323         //save the preData to a file called weather.txt
324         net.saveToFile(inputFile, preData) ;
325         //tell the user the save is complete
326         System.out.println("Saved winds aloft to file");
327      } 
328      
329   }

438   /**
439   This function is called whenever a combobox item is selected. It makes
440   sure that two combobox items, station name and altitude, must be selected in
441   order for the text fields to update with
442   correct weather information

444   @param strApt a string representing the name of the airport
445   @param strAlt a character string representing the weather altitude in thousands of
446                  feet.
447   */ 
448   public void updateTextBoxes(String strApt, String strAlt)
449   {  
450      String aptStatus = "";   //a string used for storing the airport status
451      String strKey = "" ;     //a string used for storing the icao code
452      String strStationWeather = "";   //a string used for storing a line of weather
453      String windDir;          //a string used for storing the wind direction
454      String windSpeed;        //a string used for storing the wind speed
455      String windTemp;         //a string used for storing the wind temperatur
456      String latitude;         //a string used to store the latitude
457      String longitude ;       //a string used to store the longitude
458      String surfaceTemp ;     //a string used to store the surface Temperature
459      String relativeHumidity ;//a string used to store the relative humidity
460      //Make a connection to the database to get the stationID
461      DBUpdt IDfind = new DBUpdt() ;          //create a DBUpdt object
462      IDfind.openConnection("Weather") ;      // name of connection, not of file
463      String IDcommand = "SELECT stationID FROM stations WHERE stationName = '" + strApt + "'" ;
464      //execute the above SQL command
465      IDfind.query(IDcommand) ;                  //create a query to create a result set 
466      //read the resultset to get the station ID for the selected airport in the ComboBox
467      while(IDfind.moreRecords())
468      {
469         strKey = IDfind.getField("stationID");
470         System.out.println(strKey);
471      } 
472      IDfind.close() ;//close the database so that others can use it
473      //make a connection to the database to get the weather for the airport selected in the combobox
474      DBUpdt weatherFind =new DBUpdt() ;     //create a DBUpdt object
475      weatherFind.openConnection("Weather") ;//name of connection, not of file
476      String weatherCommand ="SELECT windsaloft FROM stations WHERE stationName = '" +strApt+ "'" ;
477      //execute the above SQL statement
478      weatherFind.query(weatherCommand) ;
479      //read the resultset to get the station weather for the selected airport in the Combobox
480      while(weatherFind.moreRecords())
481      {  //weather is found in the windsaloft field
482         strStationWeather = weatherFind.getField("windsaloft");
483         System.out.println(strStationWeather);
484      } 
485      weatherFind.close() ;   //clse the database so that others can use it
486      try
487      {
488         NWSFB fb = new NWSFB(strStationWeather);  //use this NWSB object to get information for alt and airport combinations
489         //update the wind direction, speed, and temp
490         windDir = fb.getWindDirection(strAlt.substring(0,2));
491         windSpeed = fb.getWindSpeed(strAlt.substring(0,2)) ;
492         windTemp = fb.getWindTemp(strAlt.substring(0,2)) ;      
493      //create an XMLRead object to read the NWS surface weather
494         XMLRead myXMLReader = new XMLRead() ;
495      //hand the XML object the icaoCode to put in it's URL
496         String stationSurfaceWeather = "http://w1.weather.gov/xml/current_obs/" +strKey +".xml" ;
497      //Load the page
498         myXMLReader.loadPage(stationSurfaceWeather);
499         myXMLReader.setTable("current_observation");
500      //Get the latitude, longitude, surface temp, relative humidity, from the XML object
501         latitude = myXMLReader.getField("latitude"); 
502         //if latitude is negative, it is in the South
503         if (latitude.substring(0,1).compareTo("-")==0)
504         {
505            latitude=latitude.substring(1)+ " S" ;
506         }
507         //if latitude is positive, it is in the North
508         else
509         {
510            latitude=latitude + " N" ;
511         }
512         longitude = myXMLReader.getField("longitude");
513         //if longitude is negative, it is in the West,
514         if(longitude.substring(0,1).compareTo("-")==0)
515         {
516            longitude=longitude.substring(1)+ " W" ;
517         }
518         //if longitude is positive, it is in the East,
519         else
520         {
521            longitude=longitude + " E" ;
522         }
523         surfaceTemp = myXMLReader.getField("temp_c") + " C";
524         relativeHumidity = myXMLReader.getField("relative_humidity")+" %";
