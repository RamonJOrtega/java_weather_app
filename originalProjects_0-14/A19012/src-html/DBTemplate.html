009
010public interface DBTemplate
011{
012        /**
013   This function adds a record to the specified table using the specified
014   key. It returns true if the record was added and false if not.
015   
016   @param strTable a string representing the name of the table in the database
017   @param strKeyName a string representing the name of the key field in the table
018   @param strKeyContents a string representing the contents to be placed in the key field
019   @return blnStatus a boolean flag representing whether or not the record was added
020   */ 
021        public boolean addRecord(String strTable, String strKeyName, String strKeyContents) ;
022   /**
023   This function sets the db connection to null, therby closing the connection.
024   This function is necessary for other users to access the database without problems
025   
026   @param none
027   @return none
028   */ 
029        public void close() ;
030   /**
031   This function updates the database by deleting all of the records in
032   a specific table.If the rows were deleted successfully, the function 
033   returns true and false if not.
034      
035   @param strTable a string representing the name of the table in the database
036   @return flag a boolean flag representing whether or not the records were deleted
037   */ 
038        public boolean deleteAll(String strTable) ;
039   /**
040   This function is used to retrieve data from a single field in a table. It is
041   used in conjunction with the query function to desplay the database table field
042   contents to the user.
043   
044   @param strFieldName a string representing the name of a field to be displayed
045   @return strRet a string representing the contents of the specified field
046   */ 
047        public String getField(String strFieldName) ;
048   /**
049   This function is used to tell if a result set has more records. It returns true
050   if there are more records, and false if not.
051   
052   @param none
053   @return blnRet a boolean flag representing whether or not there are more records
054   */ 
055        public boolean moreRecords() ;
056   /**
057   This function opens the database connection between java and microsoft access.
058   
059   @param strDataSourceName a string the ODBC name you gave the operating system
060                              under administrative tools
061   @return none
062   */
063        public void openConnection(String strDataSourceName) ;
064   /**
065   This function passes an SQL statement to the database and creates an internal
066   record set that can be accessed by the getField() function
067   
068   @param strSQL a string representing a valid SQL statement
069   @return none
070   */ 
071        public void query(String strSQL) ;
072   /**
073   This function updates a single field in the database table with the specified contents.
074   It returns true the field was added, and false if not.
075   
076   @param strTable a string representing the database table name
077   @param strKeyName a string representing key field name
078   @param strKeyContents a string representing key field contents
079   @param strFieldName a string representing the field name to be updated
080   @param strFieldContents a string representing the contents to be added to the field name
081   @return returnFlag a boolean flag telling whethere the update was successful or not
082   */ 
083        public boolean setField(String strTable, String strKeyName, String strKeyContents, String strFieldName, String strFieldContents) ;
084        /**
085   This function simply returns a message using System.out.println() for use in testing
086   
087   @param strVar a string representing the words you want displayed to the user
088   @return none
089   */
090   public void status(String strVar) ;
091        
092}
