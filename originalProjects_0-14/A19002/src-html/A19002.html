015public class A19002
016{
017        public static void main(String[] args)
018        {
019                //Print Black Mountain Aviation Cener
020                System.out.println("Black Mountain Aviation Center") ;
021                //Print creator's name
022                System.out.println("Programmed by Ramon J. Ortega \n") ;
023                
024                //declare variables
025                double rampWeight ;
026                double emptyWeight ;
027                double bagWeight ;
028                double gallons ;
029                int passengersFront ;
030                int passengersBack ;
031                
032                //declare constants
033                final double manWeight = 170.0 ;
034                final int fuelWeight = 6 ;
035                final int pilot = 1 ;                                   //dont forget to add the pilot as a person!
036                
037                //Create a Scanner object to read input.
038                Scanner keyboard = new Scanner(System.in);
039                                
040                //Prompt the user for Aircraft Empty Weight
041                System.out.println("Enter the Aircraft Empty Weight") ;
042                //Get Empty Weight from user
043                emptyWeight = keyboard.nextDouble();
044                
045                //Prompt the user for the Number of Passengers n the Front Seat
046                System.out.println("Enter the Number of Passengers in the Front Seat") ;
047                //Get Number of Passenters in the Front SEat from user
048                passengersFront = keyboard.nextInt() ;
049                
050                // Prompt the user for the Number of Pssengers in the Back Seat
051                System.out.println("Enter the Number of Passengers in the Back Seat") ;
052                //Get Number of Passengers in the Back Seat from user
053                passengersBack = keyboard.nextInt() ;
054                
055                // Promt the user for the Number of Gallons of Fuel
056                System.out.println("Enter the Number of Gallons of Fuel") ;
057                //Get Number of Gallons of Fuel from the user
058                gallons = keyboard.nextDouble();
059                
060                //Prompt the user for the Baggage Weight
061                System.out.println("Enter the Baggage Weight") ;
062                //Get the Baggage Weight from the user
063                bagWeight = keyboard.nextDouble() ;
064                
065                //Calculate Ramp Weight
066                rampWeight = (manWeight * (passengersFront + pilot + passengersBack)) + (fuelWeight * gallons) + (emptyWeight) + (bagWeight) ;
067                //Display the Ramp Weight
068                System.out.printf("The Ramp Weight of the Aircraft is %.2f pounds", rampWeight) ;
069
070        }
071}
