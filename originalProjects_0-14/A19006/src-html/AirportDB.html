013public class AirportDB                       //name of the class
014{
015        //Declare and initialize the array list that holds an airport weather line
016        private ArrayList<String> aptList = new ArrayList<String>() ;
017        //Declare and initialize the array list that holds the aiport ID as an index
018        private ArrayList<String> aptIndex = new ArrayList<String>() ;
019        
020        /**
021        This is the constructor for the AirportDB class. It reads an input file
022        from the user and stores the airport weather and airport index in two
023        respective array lists.
024          
025   @param inputFile a scanner object containing national weather report
026   */
027        public AirportDB (Scanner inputFile)            //Constructor
028        {
029                while( inputFile.hasNext() )                            //read the file line by line until the end
030                {
031                        String line = inputFile.nextLine() ;//store the next line of the file in a string variable
032                        
033                        //only lines of greater than 65 characters are airport weather lines
034                        if (line.length() > 65 ) //if the next line of the file is an airport weather line
035                        {
036                        aptList.add(line) ;                                             //store the line into the aptList array list
037                        
038                        aptIndex.add(line.substring(0,3)) ;     //store the first 3 characters of the line into the
039                        }                                                                                               //aptIndex array list
040                }
041        }
042        /**
043   This function returns the number of airports in the aptIndex array list
044   for the user.
045   
046   @param none
047   @return an integer representing the number of airports in the aptIndex array list
048   */
049        public int size() 
050        {
051                return aptIndex.size () ;              //use the size methode for array lists
052        }
053   /**
054   This function returns the number of airports in the aptIndex array list
055   for the user.
056   
057   @param strStationID a 3 letter string representing an airport code
058   @return a string representing the full station weather
059   */
060        public String getStationWeather( String strStationID) 
061                {
062                        String ID = strStationID.toUpperCase();
063                        return aptList.get(aptIndex.indexOf(ID)) ;
064                }
065}
